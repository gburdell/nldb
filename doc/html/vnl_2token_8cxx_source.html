00043     Token::toBits(PTArray<bool> &bits) const {
00044         string s = getText();
00045         ASSERT_TRUE('\'' == s[0]);
00046         string val = s.substr(2);
00047         bool ok = true;
00048         switch (s[1]) {
00049             case 'b': case 'h': case 'o':
00050                 break;
00051             case 'd':
00052             { //convert to hex
00053                 stringstream ss(val);
00054                 long unsigned n;
00055                 ss >> n;
00056                 stringstream oss;
00057                 oss << hex << n;
00058                 ok = !(ss.fail() || oss.fail());
00059                 if (ok) {
00060                     val = oss.str();
00061                     s[1] = 'h';
00062                 }
00063             }
00064                 break;
00065             default:
00066                 ok = false;
00067         }
00068         char c;
00069         unsigned k = 0;
00070         for (int i = val.length() - 1; ok && (i >= 0); i--, k++) { //lsb->msb
00071             c = val[i];
00072             if (s[1] == 'b') {
00073                 bits[k] = (c == '1'); //true|false
00074             } else {
00075                 int nbit;
00076                 unsigned u;
00077                 stringstream ss;
00078                 if (s[1] == 'h') {
00079                     ss << hex << c;
00080                     nbit = 4;
00081                 } else {
00082                     ss << oct << c;
00083                     nbit = 3;
00084                 }
00085                 ss >> u;
00086                 ok = !ss.fail();
00087                 if (ok) {
00088                     unsigned mask = 0x01;
00089                     for (int j = 0; j < nbit; j++, mask <<= 1) {
00090                         bits[k] = ((u & mask) != 0);
00091                     }
00092                 }
00093             }
00094         }
00095         for (; ok && (k < bits.length()); k++) {
00096             bits[k] = false;
00097         }
00098         return ok;
00099     }
