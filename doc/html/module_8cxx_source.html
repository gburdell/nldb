00197     wlink(Module &m, Module::trc_unresolvedCntByName &unresolved,
00198             const t_libs &libs, unsigned strength, unsigned &cnt) {
00199         typedef vnl::t_libs::const_iterator t_libIter;
00200         if (0 == cnt) {
00201             stDidModuleNames.clear();
00202         }
00203         cnt++;
00204         //detect loop
00205         const string nm = m.getName();
00206         ASSERT_TRUE(!mapHasKey(stDidModuleNames, nm));
00207         stDidModuleNames[nm] = true;
00208         Module::t_cellsByName::iterator i = m.getCellsByName().begin(),
00209                 end = m.getCellsByName().end();
00210         for (; i != end; ++i) {
00211             TRcCell &inst = i->second;
00212             const string &refnm = inst->getRefName();
00213             if (!inst->isResolved()) {
00214                 TRcModule ref;
00215                 for (t_libIter iter = libs.begin(); (iter != libs.end()) && ref.isNull(); ++iter) {
00216                     const TRcLibrary &lib = *iter;
00217                     if (lib.isValid()) {
00218                         ref = lib->getModule(refnm);
00219                     }
00220                 }
00221                 if (ref.isValid()) {
00222                     TRcObject refo = upcast(ref);
00223                     inst->setRef(refo);
00224                     if (ref->isHierarchical() && !mapHasKey(stDidModuleNames, refnm)) {
00225                         wlink(ref.asT(), unresolved, libs, strength, cnt);
00226                     }
00227                 } else {
00228                     if (unresolved.isNull()) {
00229                         unresolved = new Module::t_unresolvedCntByName();
00230                     }
00231                     if (!mapHasKey(unresolved.asT(), refnm)) {
00232                         unresolved.asT()[refnm] = 0;
00233                     }
00234                     unresolved.asT()[refnm]++;
00235                 }
00236             }
00237         }
00238     }
