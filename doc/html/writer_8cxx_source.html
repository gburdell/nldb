00225         replBitVal(deque<string> &vals) {
00226             //roll-em off into n'bxxx
00227             string peek;
00228             string nTicVal;
00229             while (!vals.empty()) {
00230                 peek = vals.back(); //peek last
00231                 if ((4 == peek.length()) && (peek.substr(0, 3) == "1'b")) {
00232                     nTicVal = peek.at(3) + nTicVal; //prepend
00233                     vals.pop_back();
00234                 } else {
00235                     break; //while
00236                 }
00237             }
00238             unsigned n = nTicVal.size();
00239             ASSERT_TRUE(0 < n);
00240             ostringstream os;
00241             os << n << "'b" << nTicVal;
00242             vals.push_back(os.str());

00258         static
00259         bool
00260         reduceBitVec(const t_asgnEle &expr, deque<string> &rval,
00261                 const unsigned startI, const unsigned currI, bool force = false) {
00262             ASSERT_TRUE(startI <= currI);
00263             bool reduce = force || (startI == currI);
00264             unsigned endI = currI;
00265             if (!reduce) {
00266                 //consecutive bits?
00267                 const TRcWireBitRef &prev = WireBitRef::downcast(expr[currI - 1]);
00268                 const TRcWireBitRef &curr = WireBitRef::downcast(expr[currI]);
00269                 reduce = (prev->getBusName() != curr->getBusName());
00270                 int delta = curr->getBitIx() - prev->getBitIx();
00271                 reduce |= (delta != curr->getBusIncr());
00272                 if (reduce) {
00273                     endI--; //dont include curr bit
00274                 }
00275             }
00276             if (reduce) {
00277                 const TRcWireBitRef &start = WireBitRef::downcast(expr[startI]);
00278                 const TRcWireBitRef &end = WireBitRef::downcast(expr[endI]);
00279                 const unsigned ixs[2] = {start->getBitIx(), end->getBitIx()};
00280                 ostringstream os;
00281                 os << start->getBusName();
00282                 if ((ixs[0] != start->getBus()->getLb()) ||
00283                         (ixs[1] != start->getBus()->getRb())) {
00284                     os << "[" << ixs[0];
00285                     if (ixs[1] != ixs[0]) {
00286                         os << ":" << ixs[1];
00287                     }
00288                     os << "]";
00289                 }
00290                 string s = os.str();
00291                 rval.push_back(s);
00292             }
00293             return reduce;
00294         }

00296         /*
00297          * Compress {...} expression.
00298          * 4 cases:
00299          * ========
00300          * 1) 1'b[01] ...
00301          * 2) foo,foo,... => {n{foo}}  (iff compression setting >1)
00302          * 3) bar[n],bar[n-1],bar[n-2],... => bar[n:n-m] (iff bit order ok)
00303          * 4) a,b,c,... => a,b,c,...
00304          */
00305         Impl&
00306         compress(const t_asgnEle &expr, vector<string> &rval) {
00307             if (expr.empty()) return *this;
00308             static const TRcWire nilw = new Wire("");
00309             static const TRcObject nilo = upcast(nilw);
00310             static const string nulNm = " "; //a null connection
00311             /* push the bogus entry on so we can iterate w/o worrying about
00312              * the end condition.
00313              */
00314             const_cast<t_asgnEle&> (expr).push_back(nilo);
00315 
00316             enum CompressState {
00317                 eWait, eTic01, eBit
00318             };
00319             CompressState state = eWait;
00320             deque<string> tmpRval;
00321             string curr;
00322             unsigned spanStartI = 0;
00323             unsigned typeId;
00324             TRcWire wire;
00325             for (unsigned i = 0; i < expr.size(); ++i) {
00326                 const TRcObject &ele = expr[i];
00327                 if (ele.isValid()) {
00328                     curr = ele->getName();
00329                     typeId = ele->getTypeId(); //Wire, Port, WireBitRef
00330                 } else {
00331                     curr = nulNm;
00332                     typeId = 0;
00333                 }
00334                 if (0 == m_config->m_compressBusConnExpr) {
00335                     //keep bit blasted
00336                     tmpRval.push_back(curr);
00337                     continue;
00338                 }
00339                 if (WireBitRef::stTypeId != typeId) {
00340                     wire = Wire::downcast(ele);
00341                 }
00342                 switch (state) {
00343                     case eWait:
00344                         if (WireBitRef::stTypeId == typeId) {
00345                             state = eBit;
00346                             spanStartI = i; //mark start in expr[]
00347                         } else {
00348                             tmpRval.push_back(curr);
00349                             if (wire->isConst()) {
00350                                 state = eTic01;
00351                             }
00352                             //else state=eWait;
00353                         }
00354                         break;
00355                     case eTic01:
00356                         //at least 1 1'b[01] is on tail
00357                         if ((WireBitRef::stTypeId == typeId) || !wire->isConst()) {
00358                             //not a bit, so crunch existing n'bxxx
00359                             replBitVal(tmpRval);
00360                             if (WireBitRef::stTypeId == typeId) {
00361                                 state = eBit;
00362                                 spanStartI = i;
00363                             } else {
00364                                 tmpRval.push_back(curr);
00365                                 state = eWait;
00366                             }
00367                         } else {
00368                             ASSERT_TRUE(wire->isConst());
00369                             tmpRval.push_back(curr);
00370                             //state=eTic01;
00371                         }
00372                         break;
00373                     case eBit:
00374                         if (WireBitRef::stTypeId == typeId) {
00375                             if (reduceBitVec(expr, tmpRval, spanStartI, i)) {
00376                                 spanStartI = i; //reset to this bit
00377                             }
00378                             // else state=eBit
00379                         } else {
00380                             //not a bit, so reduce [spanStarI,i-1]
00381                             ASSERT_TRUE(reduceBitVec(expr, tmpRval, spanStartI, i - 1, true));
00382                             tmpRval.push_back(curr);
00383                             state = (wire->isConst()) ? eTic01 : eWait;
00384                         }
00385                         break;
00386                     default:
00387                         ASSERT_NEVER;
00388                 }
00389             }
00390             //Remove bogus entries: remove nilo
00391             ASSERT_TRUE(tmpRval.back().empty()); //nilo is ""
00392             tmpRval.pop_back();
00393             const_cast<t_asgnEle&> (expr).pop_back();
00394             //
00395             rval.clear();
00396             if (1 < m_config->m_compressBusConnExpr) {
00397                 //push bogus to simplify end condition handling
00398                 static const string nil = "";
00399                 tmpRval.push_back(nil);
00400                 //handle repeated
00401                 unsigned n = 1;
00402                 string last;
00403                 for (unsigned i = 0; i < tmpRval.size(); ++i) {
00404                     string vi = tmpRval[i];
00405                     if (0 == i) {
00406                         last = vi;
00407                     } else {
00408                         const bool match = (last == vi) && (vi != nulNm);
00409                         if (match) {
00410                             n++;
00411                         } else {
00412                             ostringstream os;
00413                             if (1 < n) {
00414                                 os << '{' << n << '{' << last << "}}";
00415                             } else {
00416                                 os << last;
00417                             }
00418                             string s = os.str();
00419                             rval.push_back(s);
00420                             if (!match) {
00421                                 last = vi;
00422                                 n = 1;
00423                             }
00424                         }
00425                     }
00426                 }
00427                 ASSERT_TRUE(!rval.back().empty()); //make sure nil didnt slip by
00428             } else {
00429                 FOREACH(deque<string>::const_iterator, tmpRval.begin(), tmpRval.end(), rval.push_back);
00430             }
00431             return *this;
00432         }
