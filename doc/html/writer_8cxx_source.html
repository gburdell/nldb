00248         reduceBitVec(const t_asgnEle &expr, deque<string> &rval,
00249                 const unsigned startI, const unsigned currI, bool force = false) {
00250             ASSERT_TRUE(startI <= currI);
00251             bool reduce = force || (startI == currI);
00252             unsigned endI = currI;
00253             if (!reduce) {
00254                 //consecutive bits?
00255                 const TRcWireBitRef &prev = WireBitRef::downcast(expr[currI - 1]);
00256                 const TRcWireBitRef &curr = WireBitRef::downcast(expr[currI]);
00257                 reduce = (prev->getBusName() != curr->getBusName());
00258                 int delta = curr->getBitIx() - prev->getBitIx();
00259                 reduce |= (delta != curr->getBusIncr());
00260                 if (reduce) {
00261                     endI--; //dont include curr bit
00262                 }

00267                 const unsigned ixs[2] = {start->getBitIx(), end->getBitIx()};
00268                 ostringstream os;

00270                 if ((ixs[0] != start->getBus()->getLb()) ||
00271                         (ixs[1] != start->getBus()->getRb())) {
00272                     os << "[" << ixs[0];
00273                     if (ixs[1] != ixs[0]) {
00274                         os << ":" << ixs[1];
00275                     }
00276                     os << "]";
00277                 }
00278                 string s = os.str();
00279                 rval.push_back(s);
00280             }

00284         /*
00285          * Compress {...} expression.
00286          * 4 cases:
00287          * ========
00288          * 1) 1'b[01] ...
00289          * 2) foo,foo,... => {n{foo}}  (iff compression setting >1)
00290          * 3) bar[n],bar[n-1],bar[n-2],... => bar[n:n-m] (iff bit order ok)
00291          * 4) a,b,c,... => a,b,c,...
00292          */
00293         Impl&
00294         compress(const t_asgnEle &expr, vector<string> &rval) {
00295             if (expr.empty()) return *this;
00296             static const TRcWire nilw = new Wire("");
00297             static const TRcObject nilo = upcast(nilw);
00298             static const string nulNm = " "; //a null connection
00299             /* push the bogus entry on so we can iterate w/o worrying about
00300              * the end condition.
00301              */
00302             const_cast<t_asgnEle&> (expr).push_back(nilo);
00303 
00304             enum CompressState {
00305                 eWait, eTic01, eBit
00306             };
00307             CompressState state = eWait;
00308             deque<string> tmpRval;
00309             string curr;
00310             unsigned spanStartI = 0;
00311             unsigned typeId;
00312             TRcWire wire;
00313             for (unsigned i = 0; i < expr.size(); ++i) {
00314                 const TRcObject &ele = expr[i];
00315                 if (ele.isValid()) {
00316                     curr = ele->getName();
00317                     typeId = ele->getTypeId(); //Wire, Port, WireBitRef
00318                 } else {
00319                     curr = nulNm;
00320                     typeId = 0;
00321                 }
00322                 if (0 == m_config->m_compressBusConnExpr) {
00323                     //keep bit blasted
00324                     tmpRval.push_back(curr);
00325                     continue;
00326                 }
00327                 if (WireBitRef::stTypeId != typeId) {
00328                     wire = Wire::downcast(ele);
00329                 }
00330                 switch (state) {
00331                     case eWait:
00332                         if (WireBitRef::stTypeId == typeId) {
00333                             state = eBit;
00334                             spanStartI = i; //mark start in expr[]
00335                         } else {
00336                             tmpRval.push_back(curr);
00337                             if (wire->isConst()) {
00338                                 state = eTic01;
00339                             }
00340                             //else state=eWait;
00341                         }
00342                         break;
00343                     case eTic01:
00344                         //at least 1 1'b[01] is on tail
00345                         if ((WireBitRef::stTypeId == typeId) || !wire->isConst()) {
00346                             //not a bit, so crunch existing n'bxxx
00347                             replBitVal(tmpRval);
00348                             if (WireBitRef::stTypeId == typeId) {
00349                                 state = eBit;
00350                                 spanStartI = i;
00351                             } else {
00352                                 tmpRval.push_back(curr);
00353                                 state = eWait;
00354                             }
00355                         } else {
00356                             ASSERT_TRUE(wire->isConst());
00357                             tmpRval.push_back(curr);
00358                             //state=eTic01;
00359                         }
00360                         break;
00361                     case eBit:
00362                         if (WireBitRef::stTypeId == typeId) {
00363                             if (reduceBitVec(expr, tmpRval, spanStartI, i)) {
00364                                 spanStartI = i; //reset to this bit
00365                             }
00366                             // else state=eBit
00367                         } else {
00368                             //not a bit, so reduce [spanStarI,i-1]
00369                             ASSERT_TRUE(reduceBitVec(expr, tmpRval, spanStartI, i - 1, true));
00370                             tmpRval.push_back(curr);
00371                             state = (wire->isConst()) ? eTic01 : eWait;
00372                         }
00373                         break;
00374                     default:
00375                         ASSERT_NEVER;
00376                 }
00377             }
00378             //Remove bogus entries: remove nilo
00379             ASSERT_TRUE(tmpRval.back().empty()); //nilo is ""
00380             tmpRval.pop_back();
00381             const_cast<t_asgnEle&> (expr).pop_back();
00382             //
00383             rval.clear();
00384             if (1 < m_config->m_compressBusConnExpr) {
00385                 //push bogus to simplify end condition handling
00386                 static const string nil = "";
00387                 tmpRval.push_back(nil);
00388                 //handle repeated
00389                 unsigned n = 1;
00390                 string last;
00391                 for (unsigned i = 0; i < tmpRval.size(); ++i) {
00392                     string vi = tmpRval[i];
00393                     if (0 == i) {
00394                         last = vi;
00395                     } else {
00396                         const bool match = (last == vi) && (vi != nulNm);
00397                         if (match) {
00398                             n++;
00399                         } else {
00400                             ostringstream os;
00401                             if (1 < n) {
00402                                 os << '{' << n << '{' << last << "}}";
00403                             } else {
00404                                 os << last;
00405                             }
00406                             string s = os.str();
00407                             rval.push_back(s);
00408                             if (!match) {
00409                                 last = vi;
00410                                 n = 1;
00411                             }
00412                         }
00413                     }
00414                 }
00415                 ASSERT_TRUE(!rval.back().empty()); //make sure nil didnt slip by
00416             } else {
00417                 FOREACH(deque<string>::const_iterator, tmpRval.begin(), tmpRval.end(), rval.push_back);
00418             }
00419             return *this;
00420         }
