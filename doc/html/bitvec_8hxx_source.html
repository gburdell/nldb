00330 template<typename T>
00331 T
00332 PTBitVecMethods<T>::assign(T& lhs, const TBvBase& bref, 
00333                                                    const T& v,
00334                                                    unsigned lb, unsigned rb)
00335 {
00336         bref.checkBounds(lb, rb);
00337         T val = lhs;
00338         // zero out bit slots we will set
00339         unsigned offset = bref.offset(rb);
00340         ASSERT_TRUE(offset < stMaxBits);
00341         T mask = onesMask(lb, rb) << offset; // mask <= 1s covering [lb,rb]
00342         val &= ~mask;    // val <= zero out [lb,rb]
00343         T v2 = v;
00344         v2 <<= offset;  // ...v000
00345         v2 &= mask;
00346         val |= v2;
00347         lhs = val;
00348         return valueAt(lhs, bref, lb, rb);
