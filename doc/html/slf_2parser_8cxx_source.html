00151         TRcToken libnmTok;
00152         try {
00153             while (la(0)->getType() == Token::eLibrary) {
00154                 accept();

00156                 libnmTok = la(0);

00158                     case Token::eIdent: case Token::eString:

00170                     pair<TRcLibraryEle, TRcValueSet> lset = libraryEle(lib);
00171                     TRcLibraryEle &lele = lset.first;
00172                     if (lele.isValid()) {
00173                         //TODO: put cells in lib and otherstuff???
00174                         if (LibraryEle::eKeyValue == lele->getType()) {
00175                             //Look for these keys
00176                             if (lele->getKey() == cType) {
00177                                 TRcKeyValue asKv = KeyValue::downcast(lele);
00178                                 addTypeGroup(m_libBusTypes, asKv);
00179                             }
00180                         } else {
00181                             ASSERT_TRUE(LibraryEle::eLibCell == lele->getType());

00300     Parser::valueType() throw (unsigned) {
00301         TRcValueType valt;
00302         Token::EType types[2] = {la(0)->getType(), la(1)->getType()};
00303 
00304         /*:   ((IDENT|number) expr_op)=> expr
00305               |   kident (bus)?
00306               |   STRING
00307               |   number (options{greedy=true;}: unit)?
00308               |   bool         
00309          */
00310         if (((Token::eIdent == types[0]) || (Token::eFloat == types[0]) || (Token::eInteger == types[0]))
00311                 && (Token::eStar == types[1] || Token::ePlus == types[1])) {
00312             TRcExpr ex = expr();
00313             valt = new ValueType(ex);
00314         } else {
00315             switch (types[0]) {
00316                 case Token::eIdent: case Token::eCell: case Token::eLibrary: //kident
00317                 {
00318                     TRcToken kident = accept();
00319                     TRcBus bs;
00320                     if (Token::eLBracket == types[1]) {
00321                         bs = bus();
00322                     }
00323                     valt = new ValueType(kident->getText(), bs);
00324                 }
00325                     break;
00326                 case Token::eString:
00327                 {
00328                     TRcToken str = accept();
00329                     valt = new ValueType(rmEnclQuotes(str->getText()));
00330                 }
00331                     break;
00332                 case Token::eInteger: case Token::eFloat:
00333                 {
00334                     TRcNumber num = number();
00335                     TRcToken unit;
00336                     if (Token::eIdent == types[1]) {
00337                         unit = accept();
00338                     }
00339                     string us = (unit.isNull()) ? "" : unit->getText();
00340                     valt = new ValueType(num, us);
00341                 }
00342                     break;
00343                 case Token::eTrue: case Token::eFalse:
00344                 {
00345                     TRcToken boolVal = accept();
00346                     valt = new ValueType(boolVal->getText() == cTrue);
00347                 }
00348                     break;
00349                 default:
00350                 {
00351                     TRcToken bad = la(0);
00352                     error2(bad);
00353                     throw (++m_errCnt);
00354                 }
00355             }
00356         }

00464     Parser::createLibCell(TRcLibrary &lib, TRcLibCell &lcel, TRcValueSet &rcvset) {

00470         ASSERT_TRUE(!lib->hasModule(lcel->getName()));
00471         {
00472             vnl::TRcModule mod = asModule(lcel);
00473             lib->add(mod);
00474         }
00475         //Need to walk linearly, since need to retain port order
00476         //and can get repeat keys, like bus,...
00477         const ValueSet &vset = rcvset.asT();
00478         for (unsigned i = 0; i < vset.length(); i++) {
00479             const TRcKeyValue &kv = vset[i];
00480             const string &key = kv->getKey();
00481             if (key == cArea) {
00482                 lcel->setArea(kv->getVal()->asNumber()->asDouble());
00483             } else if ((key == cBus) || (key == cPin)) {
00484                 const string &portNm = kv->getVal()->asIdent();
00485                 ASSERT_TRUE(kv->hasValSet());
00486                 //get as map, allow dups, since bus expanded into pin(s)
00487                 ValueSet::trc_kvByKey rcvbk = kv->getValSet()->asMap(true);
00488                 ValueSet::t_kvByKey &vbk = rcvbk.asT();
00489 #ifdef xDEBUG
00490                 dbgOs << "DBG: pin(" << portNm << ") {" << endl;
00491                 dbgOs << kv->getVal() << endl;
00492                 dbgOs << kv->getValSet() << " }" << endl;
00493 #endif
00494                 string direction;
00495                 TRcKeyValue dirkv;
00496                 if (!mapGetVal(rcvbk, cDirection, dirkv)) {
00497                     //get direction from bus(xxx) { pin(yyy) { direction : ..}}
00498                     ASSERT_TRUE((cBus == key) && mapGetVal(rcvbk, cPin, dirkv));
00499                     ASSERT_TRUE(mapGetVal(dirkv->getValSet()->asMap(true), cDirection, dirkv));
00500                 }
00501                 direction = dirkv->getVal()->asIdent();
00502                 if (cBus == key) {
00503                     ASSERT_TRUE(m_libBusTypes.isValid());
00504                     const string &busType = vbk[cBusType]->getVal()->asIdent();

00506                     TRcBus bus = m_libBusTypes.asT()[busType];

00508                     //TODO: bussed pin (bit-blasted) funcs and timing
